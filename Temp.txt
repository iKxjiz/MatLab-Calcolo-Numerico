npt = 200;
t = linspace(a, b, npt);
Cxy = ppbezier_val(ppC, t);
[x, y] = cp2_circle(t);
Qxy = [x', y'];
MaxErr = max(vecnorm((Cxy - Qxy)'));
fprintf("Massimo errore: %e\n", MaxErr);

bool = MaxErr <= 2.0e-03;
if bool == 1
    fprintf("L'errore è contenuto");
else
    fprintf("L'errore NON è contenuto");
end
